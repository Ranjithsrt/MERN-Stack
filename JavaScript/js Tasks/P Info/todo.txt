// script.js
// This file contains the main JavaScript logic for the web page.
// Author: [Your Name]
// Last updated: 2025-09-24
// Description: Handles DOM interactions and page functionality.

const contactForm = document.getElementById("contactForm");
const resultList = document.querySelector("#result ul");

let contacts = []; // store submitted contacts

contactForm.addEventListener("submit", function (event) {
  event.preventDefault(); // prevent form submission

  const name = document.getElementById("name").value.trim();
  const contact = document.getElementById("contact").value.trim();
  const email = document.getElementById("email").value.trim();

  // Fixed validation - check for empty strings after trimming
  if (name === "" || contact === "" || email === "") {
    alert("Please fill all the details");
    return;
  }

  // Additional email format validation (optional)
  // const emailPattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  // if (!emailPattern.test(email)) {
  //   alert("Please enter a valid email address");
  //   return;
  // }

  const newContact = { id: Date.now(), name, contact, email };
  contacts.push(newContact);
  renderContacts();
  contactForm.reset(); // clear form fields
});

function renderContacts() {
  resultList.innerHTML = ""; // clear previous results
  contacts.forEach((contact) => {
    const li = document.createElement("li");
    li.style.display = "flex";
    li.style.gap = "20px";
    li.innerHTML = `
      <span>${contact.name}</span>
      <span>${contact.contact}</span>
      <span>${contact.email}</span>
      <button onclick="editContact(${contact.id})">Edit</button>
      <button onclick="deleteContact(${contact.id})">Delete</button>
    `;
    resultList.appendChild(li);
  });
}

function editContact(id) {
  const contact = contacts.find((c) => c.id === id);
  if (contact) {
    document.getElementById("name").value = contact.name;
    document.getElementById("contact").value = contact.contact;
    document.getElementById("email").value = contact.email;
    deleteContact(id); // remove the old entry
  }
}

function deleteContact(id) {
  contacts = contacts.filter((c) => c.id !== id);
  renderContacts();
}

renderContacts(); // initial render
